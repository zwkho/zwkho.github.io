    if x < -1:
        return 0
    elif -1 <= x < 0:
        return 1+x
    elif 0 <= x < 1:
        return 1-x
    elif x > 1:
        return 0

fig, ax = plt.subplots()
ax.set_xlim((-1, 8))
ax.set_ylim((0, 1))

line, = ax.plot([], [], lw=2)

def init():
    line.set_data([], []);
    return (line,)

def animate(i):
    v = 0.01
    x = np.linspace(-1, 8, 1000)
    y = [fun(xx - v*i) for xx in x]
    line.set_data(x, y)
    return (line,)

anim = animation.FuncAnimation(fig, animate, init_func=init, frames=500, interval=20, blit=True, repeat = False)
